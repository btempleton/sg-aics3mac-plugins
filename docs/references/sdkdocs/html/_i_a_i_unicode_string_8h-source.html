01136 inline UnicodeString& UnicodeString::insert (size_type insertOffset, const UnicodeString& str,
01137                 size_type offset, size_type count)
01138 {
01139         if ( insertOffset > length() || offset > str.length() )
01140                 throw ai::Error(kUnicodeStringBadIndex);
01141         UnicodeString result = substr(0, insertOffset);
01142         result.append(str, offset, count);
01143         result.append(substr(insertOffset));
01144 
01145         *this = result;
01146         return *this;
01147 }

01149 inline UnicodeString& UnicodeString::replace (size_type pos, size_type num, const UnicodeString& str,
01150                                                 size_type startOffset, size_type count)
01151 {
01152         if ( pos > length() || startOffset > str.length() )
01153                 throw ai::Error(kUnicodeStringBadIndex);
01154         erase(pos, num);
01155         insert(pos, str, startOffset, count);
01156 
01157         return *this;
01158 }

01175 inline UnicodeString::size_type UnicodeString::as_ASUnicode ( ASUnicode* buffer, size_type bufferMax ) const
01176 {
01177         const UTF16Char* bufPtr = 0;
01178         const size_type kThisUTF16Len = utf_16(bufPtr) + 1;
01179         const size_type kCopyMax = (bufferMax < kThisUTF16Len ? bufferMax : kThisUTF16Len) - 1;
01180         memcpy(buffer, bufPtr, kCopyMax*sizeof(UTF16Char));
01181         buffer[kCopyMax] = 0;
01182 
01183         return kThisUTF16Len;
01184 }

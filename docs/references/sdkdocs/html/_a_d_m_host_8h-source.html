00073 // -----------------------------------------------------------------------------
00074 //      GetADMVersion returns the version number ADM is compiled with. This number
00075 //      consists of a hi word which is the major version number, and a low word which
00076 //      is the minor revision number. Beginning with 2.30, the minor version is in
00077 //      the low word, high byte. The low word, low byte is for branch versions.
00078 //      
00079 //      The macro ADM_VERSION can be used to make sure you are running the version
00080 //      you think you are.
00081 
00083 #define ADM_VERSION (0x00 | (ADMVMajor << 16) | (ADMVMinor << 8) | (ADMVBuildNumber))
